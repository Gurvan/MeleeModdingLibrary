matrixExample.c

This demonstrates the basic uses of matrix.h

#include <gctypes.h> // u32
#include <matrix.h>
// declare two vectors
Vector v1 = {NULL, 0};
Vector v2 = {NULL, 0};
// declare matrix
Matrix m1 = {NULL, 0, 0};
// function used to populate vector
float popVecFunc(u32 index)
{
return 2 * index;
}
// function used to populate matrix
float popMatFunc(u32 row, u32 col)
{
return row + col;
}
void _main()
{
// initialize both vectors to have 10 elements (all zero)
initVector(&v1, 10);
initVector(&v2, 10);
// populate v2 with values
populateVec(&v2, &popVecFunc);
// take dot product of v1 and v2
float dProd = vectorDotProduct(&v1, &v2);
// initialize 10x10 matrix and populate it
initMatrix(&m, 10, 10);
populateMat(&m, &popMatFunc);
// multiple m and v2, store in v1
matrixVectorProdut(&v1, &m, &v2);
}
